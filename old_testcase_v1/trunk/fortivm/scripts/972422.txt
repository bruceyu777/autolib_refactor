# 972422
# Verify ev license, support admin vdom, not multi-vdom

[FVM_A]

    diagnose debug vm-print-license
    expect -e "Model: EVAL" -for 972422 -t 5
    expect -e "permanent: 2" -for 972422 -t 5

    config system global
        set vdom-mode multi-vdom
    end
    y

[FVM_A]
    config vdom
       edit vdom1
    
    get system status | grep "Virtual domain"
    expect -e "1 in NAT mode, 0 in TP mode" -for 972422 -t 5
    expect -e "Virtual domain configuration: multiple" -for 972422 -t 5

    config vdom
    edit root
        config system settings
            set vdom-type admin
        end
        y
    end

   config vdom
   edit vdom1
   end

   get system status | grep "Virtual domain"
   expect -e "2 in NAT mode, 0 in TP mode" -for 972422 -t 5

   config vdom
   edit vdom2
   get system status | grep "Virtual domain"
   expect -e "2 in NAT mode, 0 in TP mode" -for 972422 -t 5

   config vdom
   delete vdom1
   
   edit root
        config system settings
            set vdom-type traffic
        end
        y
    end

    config global
    config system global
        set vdom-mode multi-vdom
    end
    y

[FVM_A]
    get system status
    report 972422
